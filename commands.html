        # Пример 1: создание переменных
        x = 10        # целое число
        y = 3.14      # число с плавающей точкой
        name = "Python"  # строка
        print(x, y, name)

        

        # Пример 2: условный оператор
        age = 18
        if age >= 18:
            print("Вам 18 лет или больше")
        else:
            print("Вы моложе 18 лет")

        

        # Пример 3: цикл for
        for i in range(5):  # от 0 до 4
            print(i)

        

        # Пример 4: цикл while
        count = 0
        while count < 5:
            print(count)
            count += 1  # увеличиваем count на 1

        

        # Пример 5: функция
        def greet(name):
            print(f"Привет, {name}!")

        greet("Иван")

        

        # Пример 6: список
        fruits = ["яблоко", "банан", "киви"]
        fruits.append("апельсин")  # добавление элемента в список
        print(fruits)

        

        # Пример 7: словарь
        person = {"name": "Иван", "age": 30}
        print(person["name"])  # доступ к значению по ключу

        

        # Пример 8: множество
        numbers = {1, 2, 3, 3, 4}
        numbers.add(5)  # добавление элемента
        print(numbers)

        

        # Пример 9: обработка ошибок
        try:
            x = 10 / 0  # деление на ноль
        except ZeroDivisionError:
            print("Ошибка: деление на ноль!")

        

        # Пример 10: итерация по строкам
        text = "Python"
        for char in text:
            print(char)

        

        # Пример 11: lambda
        multiply = lambda x, y: x * y
        print(multiply(2, 3))

        

        # Пример 12: классы и объекты
        class Car:
            def __init__(self, brand, model):
                self.brand = brand
                self.model = model

            def display_info(self):
                print(f"{self.brand} {self.model}")

        car1 = Car("Toyota", "Corolla")
        car1.display_info()

        

        # Пример 13: работа с файлами
        with open("example.txt", "w") as file:
            file.write("Привет, мир!")

        with open("example.txt", "r") as file:
            print(file.read())

        

        # Пример 14: работа с библиотекой math
        import math
        print(math.sqrt(16))  # извлечение квадратного корня
        print(math.pi)        # число Пи

        

        # Пример 15: объединение строк
        first_name = "Иван"
        last_name = "Иванов"
        full_name = first_name + " " + last_name
        print(full_name)

        

        # Пример 16: форматирование строк
        name = "Иван"
        age = 30
        print(f"Меня зовут {name}, мне {age} лет.")

        

        # Пример 17: поиск элемента в списке
        fruits = ["яблоко", "банан", "киви"]
        if "банан" in fruits:
            print("Банан есть в списке!")
        else:
            print("Банана нет в списке.")

        

        # Пример 18: работа с датой и временем
        from datetime import datetime
        now = datetime.now()
        print("Текущая дата и время:", now)

        

        # Пример 19: использование zip
        names = ["Иван", "Мария", "Петр"]
        ages = [30, 25, 40]
        combined = zip(names, ages)

        for name, age in combined:
            print(f"{name} - {age} лет")

        

        # Пример 20: сортировка списка
        numbers = [5, 2, 9, 1, 5, 6]
        numbers.sort()  # сортировка по возрастанию
        print(numbers)
